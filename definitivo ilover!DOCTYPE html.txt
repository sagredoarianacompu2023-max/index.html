    // ESTA FUNCIÓN ES LA QUE CAMBIA LAS PESTAÑAS
    function abrir(idSeccion) {
        // 1. Buscamos todas las secciones y las ocultamos
        const secciones = document.querySelectorAll('.ventana');
        secciones.forEach(sec => {
            sec.classList.remove('visible');
        });

        // 2. Mostramos solo la sección que tiene el ID que recibimos
        const seccionAMostrar = document.getElementById(idSeccion);
        if (seccionAMostrar) {
            seccionAMostrar.classList.add('visible');
        }
    }

        const display = document.getElementById(idContenedor);

            const randomMsg = lista[Math.floor(Math.random() * lista.length)];
            display.innerText = randomMsg;
            display.style.opacity = 1;
        }, 150);
